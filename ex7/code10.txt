float x = 300;
float y = 200;
int r = 180;

void setup(){
size(600, 400);
background(255,255,255);
frameRate(60);
}

//‰~‚ğ•¡”‚©‚­
void draw(){
//c‘œ‚ğ‚È‚­‚·
fill(255);
noStroke();
rect(0, 0, 600, 400);

x = x + random(-5, 5);
y = y + random(-5, 5);

for(int i=0 ; i < 10 ; i=i+1){
noFill();
stroke(random(255), random(255), 255);
ellipse(x + random(-10, 10), y + random(-10, 10), r, r);
}
}
